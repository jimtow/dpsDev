<!DOCTYPE html>
<html>
<head>
    <title>Simple Stopwatch</title>
</head>
<body>

<h2>Stopwatch</h2>

<p id="display">00:00.0</p>

<button onclick="startStopwatch()">Start</button>
<button onclick="stopStopwatch()">Stop</button>
<button onclick="resetStopwatch()">Reset</button>

<script>
let startTime = 0;
let elapsedTime = 0;
let timer = null;

function updateDisplay() {
    const currentTime = Date.now() - startTime + elapsedTime;

    const minutes = Math.floor(currentTime / 60000);
    const seconds = Math.floor((currentTime % 60000) / 1000);
    const tenths = Math.floor((currentTime % 1000) / 100);

    document.getElementById("display").textContent =
        String(minutes).padStart(2, "0") + ":" +
        String(seconds).padStart(2, "0") + "." +
        tenths;
}

function startStopwatch() {
    if (timer !== null) return;   // Already running

    startTime = Date.now();

    timer = setInterval(updateDisplay, 100);
}

function stopStopwatch() {
    if (timer === null) return;

    clearInterval(timer);
    timer = null;

    elapsedTime += Date.now() - startTime;
}

function resetStopwatch() {
    clearInterval(timer);
    timer = null;

    startTime = 0;
    elapsedTime = 0;

    document.getElementById("display").textContent = "00:00.0";
}
</script>

</body>
</html>
If you only need the elapsed time in your JavaScript

A Stopwatch object is even simpler:

let startTime = 0;

function start() {
    startTime = Date.now();
}

function elapsedSeconds() {
    return (Date.now() - startTime) / 1000;
}

Example:

start();

// Later...
let seconds = elapsedSeconds();
console.log(seconds.toFixed(1));   // e.g. 3.7

For your rowing app, I'd recommend using performance.now() instead of Date.now() because it provides higher-resolution timing and isn't affected by changes to the system clock.

<script>
function calculate() {

    // Read values from the HTML
    const watts = Number(document.getElementById("watts").value);
    const time = Number(document.getElementById("time").value);

    // Call another function
    const work = calculateWork(watts, time);

    // Display the result
    document.getElementById("result").textContent =
        "Work = " + work + " joules";
}

function calculateWork(watts, time) {
    return watts * time;
}
</script>

Hide one section and show the other with JavaScript.

function showPage2() {
    document.getElementById("page1").style.display = "none";
    document.getElementById("page2").style.display = "block";
}

function showPage1() {
    document.getElementById("page2").style.display = "none";
    document.getElementById("page1").style.display = "block";
}
</script>

<!doctype html>
<html>
<body>
  <h1 id="timer">0</h1>
  <button onclick="start()">Start</button>
  <button onclick="stop()">Stop</button>
  <button onclick="reset()">Reset</button>

  <script>
    let seconds = 0;
    let interval = null;

    function update() {
      document.getElementById("timer").textContent = seconds;
    }

    function start() {
      if (interval) return;
      interval = setInterval(() => {
        seconds++;
        update();
      }, 1000);
    }

    function stop() {
      clearInterval(interval);
      interval = null;
    }

    function reset() {
      stop();
      seconds = 0;
      update();
    }
  </script>
</body>
