Example 2: Constants from HTML input boxes
<!DOCTYPE html>
<html>
<head>
    <title>Function Inputs</title>
</head>
<body>

Watts:
<input id="watts" type="number" value="250">

Time (seconds):
<input id="time" type="number" value="60">

<button onclick="calculate()">Calculate</button>

<p id="result"></p>

<script>
function calculate() {

    // Read values from the HTML
    const watts = Number(document.getElementById("watts").value);
    const time = Number(document.getElementById("time").value);

    // Call another function
    const work = calculateWork(watts, time);

    // Display the result
    document.getElementById("result").textContent =
        "Work = " + work + " joules";
}

function calculateWork(watts, time) {
    return watts * time;
}
</script>

</body>
</html>

Option 2: One HTML file with two "screens"

Hide one section and show the other with JavaScript.

<div id="page1">
    <h2>Page 1</h2>
    <button onclick="showPage2()">Next</button>
</div>

<div id="page2" style="display:none;">
    <h2>Page 2</h2>
    <button onclick="showPage1()">Back</button>
</div>

<script>
function showPage2() {
    document.getElementById("page1").style.display = "none";
    document.getElementById("page2").style.display = "block";
}

function showPage1() {
    document.getElementById("page2").style.display = "none";
    document.getElementById("page1").style.display = "block";
}
</script>

This is useful for simple apps and calculators.

<!doctype html>
<html>
<body>
  <h1 id="timer">0</h1>
  <button onclick="start()">Start</button>
  <button onclick="stop()">Stop</button>
  <button onclick="reset()">Reset</button>

  <script>
    let seconds = 0;
    let interval = null;

    function update() {
      document.getElementById("timer").textContent = seconds;
    }

    function start() {
      if (interval) return;
      interval = setInterval(() => {
        seconds++;
        update();
      }, 1000);
    }

    function stop() {
      clearInterval(interval);
      interval = null;
    }

    function reset() {
      stop();
      seconds = 0;
      update();
    }
  </script>
</body>
</html>
